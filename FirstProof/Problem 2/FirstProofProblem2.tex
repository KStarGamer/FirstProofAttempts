\title[Universal nonvanishing test vectors on $\GL_{n+1}\times\GL_n$]
{Universal nonvanishing Whittaker test vectors\\
for shifted local Rankin--Selberg integrals on $\GL_{n+1}\times \GL_n$}

\date{\today}

\subjclass[2020]{11F70, 22E50}
\keywords{Whittaker models, Kirillov models, newforms, conductors, Rankin--Selberg integrals, test vectors}

\begin{document}

\begin{abstract}
Let $F$ be a non-archimedean local field and fix a nontrivial additive character $\psi$ of conductor $\oo$.
Given a generic irreducible admissible representation $\Pi$ of $\GL_{n+1}(F)$, we construct for each $m\ge 0$
a Whittaker function $W_m\in \W(\Pi,\psi^{-1})$ depending only on $(\Pi,\psi,m)$ with the following property:
for every generic irreducible admissible representation $\pi$ of $\GL_n(F)$ of conductor exponent $a(\pi)\le m$,
for the normalized Whittaker newform $V\in \W(\pi,\psi)$ and for any generator $Q$ of the inverse conductor ideal
$\qq(\pi)^{-1}$, the shifted local Rankin--Selberg integral
\[
\int_{\N{n}\backslash \GL_n(F)} W_m(\diag(g,1)\,u_Q)\,V(g)\,\abs{\det g}^{\,s-\frac12}\,dg
\]
is absolutely convergent for all $s\in\mathbb C$ and equals the explicit nonzero constant
$\psi^{-1}(Q)\cdot \mathrm{vol}\!\bigl(\N{n}(\oo)\backslash K_1(\pp^m)\bigr)$.
In particular, for arbitrary $\pi$ one obtains nonvanishing by taking $m=a(\pi)$.
We also record the elementary smoothness obstruction showing that the ``exact level-$0$ mirabolic'' subgroup in $\GL_n(\oo)$
is not open for $n\ge 2$ and hence cannot support nonzero smooth test functions.
\end{abstract}

\section{Introduction}

Let $F$ be a non-archimedean local field with ring of integers $\oo$ and fix a nontrivial additive character
$\psi:F\to \mathbb C^\times$ of conductor $\oo$.
Let $\N{r}\subset \GL_r(F)$ denote the standard upper-triangular unipotent subgroup,
and write $\psi_r$ for the standard generic character of $\N{r}$ induced by $\psi$.
If $\Sigma$ is a generic irreducible admissible representation of $\GL_r(F)$, we denote by
$\W(\Sigma,\psi_r^{\pm 1})$ its $\psi_r^{\pm 1}$-Whittaker model.

For generic $\Pi$ on $\GL_{n+1}(F)$ and generic $\pi$ on $\GL_n(F)$, the local Rankin--Selberg integral
\begin{equation}\label{eq:RS-intro}
Z(W,V,s)=\int_{\N{n}\backslash \GL_n(F)} W(\diag(g,1))\,V(g)\,\abs{\det g}^{\,s-\frac12}\,dg
\end{equation}
depends on choices $W\in\W(\Pi,\psi^{-1})$ and $V\in\W(\pi,\psi)$.
The question addressed here is a shifted variant in which one inserts a unipotent element $u_Q$ determined by the conductor
of $\pi$, and asks whether one can force \emph{uniform nonvanishing} (and finiteness) for \emph{all} $s\in\mathbb C$.

\smallskip

\noindent\textbf{Problem.}
Let $\Pi$ be generic irreducible admissible on $\GL_{n+1}(F)$, realized in $\W(\Pi,\psi^{-1})$.
Let $\pi$ be generic irreducible admissible on $\GL_n(F)$, realized in $\W(\pi,\psi)$.
Let $\qq(\pi)\subset \oo$ be the conductor ideal of $\pi$ and choose a generator $Q\in F^\times$ of $\qq(\pi)^{-1}$.
Set $u_Q=I_{n+1}+Q\,E_{n,n+1}\in \GL_{n+1}(F)$.
Must there exist $W\in \W(\Pi,\psi^{-1})$ and $V\in\W(\pi,\psi)$ such that
\[
\int_{\N{n}\backslash \GL_n(F)} W(\diag(g,1)\,u_Q)\,V(g)\,\abs{\det g}^{\,s-\frac12}\,dg
\]
is finite and nonzero for all $s\in\mathbb C$?

\smallskip

We answer this affirmatively, with an explicit construction of $W$ and an explicit evaluation of the integral.
The key inputs are:
(i) a simple conjugation identity for the shift $u_Q$, and
(ii) the existence of Whittaker functions whose restriction to the Levi $\diag(\GL_n(F),1)$ is an \emph{arbitrary}
compactly supported $(\N{n},\psi_n^{-1})$-equivariant function. This latter extension statement goes back to
Gelfand--Kazhdan \cite{GK75} (see also the archimedean analogs of Jacquet and Kemarsky).

\section{Notation and basic definitions}

Let $F$ be a non-archimedean local field with ring of integers $\oo$, maximal ideal $\pp$, and residue cardinality $q$.
Fix a uniformizer $\varpi$ with $\pp=(\varpi)$, and normalize the absolute value $\abs{\cdot}$ by $\abs{\varpi}=q^{-1}$.

Fix a nontrivial additive character $\psi:F\to \mathbb C^\times$ of conductor $\oo$, i.e.
$\psi$ is trivial on $\oo$ and nontrivial on $\varpi^{-1}\oo$.
For $r\ge 1$, set $\G{r}=\GL_r(F)$ and let $\N{r}\subset \G{r}$ be the upper-triangular unipotent subgroup.
Define the standard generic character $\psi_r:\N{r}\to \mathbb C^\times$ by
\[
\psi_r(u)=\psi\!\Bigl(\sum_{i=1}^{r-1}u_{i,i+1}\Bigr).
\]
If $\Sigma$ is generic irreducible admissible on $\G{r}$, its $\psi_r^{\pm 1}$-Whittaker model is denoted
$\W(\Sigma,\psi_r^{\pm 1})$.

Write $K_r=\GL_r(\oo)$. For $n\ge 1$ and $m\ge 0$, define
\begin{equation}\label{eq:K1}
K_1(\pp^m)=\left\{\,k\in K_n:\ e_n k\equiv e_n \pmod{\pp^m}\,\right\},
\qquad e_n=(0,\dots,0,1)\in F^n.
\end{equation}
Equivalently, $k\in K_1(\pp^m)$ if and only if $k_{n,j}\in \pp^m$ for $1\le j<n$ and $k_{n,n}\in 1+\pp^m$.
Each $K_1(\pp^m)$ is an open compact subgroup of $\GL_n(F)$.

We embed $\GL_n(F)$ into $\GL_{n+1}(F)$ via
\[
\iota:\GL_n(F)\hookrightarrow \GL_{n+1}(F),\qquad \iota(g)=\diag(g,1).
\]
Let $E_{i,j}$ be the standard matrix unit.
Given $Q\in F$, set
\[
u_Q:=I_{n+1}+Q\,E_{n,n+1}\in \GL_{n+1}(F).
\]

For $W\in \W(\Pi,\psi_{n+1}^{-1})$ and $V\in \W(\pi,\psi_n)$ we consider the shifted local Rankin--Selberg integral
\begin{equation}\label{eq:RS}
Z(W,V,s;Q)=\int_{\N{n}\backslash \GL_n(F)} W(\iota(g)u_Q)\,V(g)\,\abs{\det g}^{\,s-\frac12}\,dg.
\end{equation}

\section{A smoothness obstruction for the exact level-$0$ mirabolic}

This section is not needed for the main theorem, but it clarifies why one \emph{cannot} force compactness by supporting on the
subgroup of $\GL_n(\oo)$ with last row \emph{exactly} equal to $e_n$ when $n\ge 2$.

\begin{definition}
For $n\ge 1$ define the \emph{exact mirabolic} subgroup of $K_n=\GL_n(\oo)$ by
\[
K_n^{\mathrm{mir}}:=\{\,k\in K_n:\ e_nk=e_n\,\}.
\]
\end{definition}

\begin{lemma}\label{lem:mir-not-open}
If $n\ge 2$, the subgroup $K_n^{\mathrm{mir}}$ is closed but not open in $K_n$.
Moreover,
\[
K_n^{\mathrm{mir}}=\bigcap_{m\ge 1}K_1(\pp^m),
\]
hence $K_n^{\mathrm{mir}}$ has empty interior in $K_n$.
\end{lemma}

\begin{proof}
The condition $e_nk=e_n$ is equivalent to the congruences $e_nk\equiv e_n\pmod{\pp^m}$ for all $m\ge 1$,
because $\bigcap_{m\ge 1}\pp^m=\{0\}$ in $\oo$. This gives the displayed identity.

Each $K_1(\pp^m)$ is open in $K_n$, and for $n\ge 2$ the sequence is strictly decreasing:
the congruence condition on the last row modulo $\pp^m$ becomes strictly stronger with $m$.
If the intersection were open, it would contain some $K_1(\pp^m)$; but that would force $K_1(\pp^m)=K_n^{\mathrm{mir}}$,
contradicting strict decrease. Thus $K_n^{\mathrm{mir}}$ is not open. As an intersection of closed subgroups, it is closed.
Finally, in a topological group the interior of a subgroup is itself a subgroup and hence open; thus a subgroup is open
if and only if it has nonempty interior. Therefore $K_n^{\mathrm{mir}}$ has empty interior.
\end{proof}

\begin{proposition}\label{prop:no-smooth-support}
Let $G$ be a totally disconnected locally compact group and let $H\subseteq G$ be closed with empty interior.
Then every locally constant function $f:G\to\mathbb C$ supported on $H$ is identically zero.
\end{proposition}

\begin{proof}
If $f(x)\neq 0$ for some $x$, then local constancy gives an open neighbourhood $U$ of $x$ on which $f$ is constant and nonzero.
Hence $U\subseteq \mathrm{supp}(f)\subseteq H$, contradicting that $H$ has empty interior.
\end{proof}

\begin{remark}
Lemma~\ref{lem:mir-not-open} and Proposition~\ref{prop:no-smooth-support} show that for $n\ge 2$ there is no nonzero smooth
function on $\GL_n(F)$ supported on $\N{n}\,K_n^{\mathrm{mir}}$. Consequently one cannot realize an ``exact level-$0$ mirabolic''
characteristic function as the restriction of a Whittaker function to the embedded Levi $\iota(\GL_n(F))$.
The construction below replaces $K_n^{\mathrm{mir}}$ by the open subgroup $K_1(\pp^m)$.
\end{remark}

\section{A conjugation identity for the shift $u_Q$}

\begin{lemma}\label{lem:conj}
Let $W\in \W(\Pi,\psi_{n+1}^{-1})$.
For all $g\in \GL_n(F)$ and all $Q\in F$, one has
\[
W(\iota(g)\,u_Q)=\psi^{-1}(Q\,g_{n,n})\,W(\iota(g)).
\]
\end{lemma}

\begin{proof}
Write
\[
\iota(g)\,u_Q=\bigl(\iota(g)\,u_Q\,\iota(g)^{-1}\bigr)\,\iota(g).
\]
Since $u_Q=I_{n+1}+Q E_{n,n+1}$ and $\iota(g)=\diag(g,1)$, we compute
\[
\iota(g)\,E_{n,n+1}\,\iota(g)^{-1}
=\sum_{i=1}^n g_{i,n}\,E_{i,n+1},
\]
hence
\[
\iota(g)\,u_Q\,\iota(g)^{-1}
=I_{n+1}+Q\sum_{i=1}^n g_{i,n}\,E_{i,n+1}\ \in\ \N{n+1}.
\]
The generic character $\psi_{n+1}$ depends only on the superdiagonal entries $(i,i+1)$, so among the matrices $E_{i,n+1}$
only $E_{n,n+1}$ contributes. Therefore
\[
\psi_{n+1}\!\bigl(\iota(g)\,u_Q\,\iota(g)^{-1}\bigr)=\psi(Q\,g_{n,n}).
\]
Using Whittaker equivariance $W(ux)=\psi_{n+1}^{-1}(u)W(x)$ for $u\in \N{n+1}$ yields the claim.
\end{proof}

\section{Kirillov--type extension to the Levi}

Let $r\ge 2$ and denote by $\iota:\GL_{r-1}(F)\hookrightarrow \GL_r(F)$ the standard Levi embedding $g\mapsto \diag(g,1)$.
Write $\N{r-1}\subset \GL_{r-1}(F)$ for the upper-triangular unipotent subgroup.

\begin{definition}
Let $r\ge 2$.
Denote by $C_c^\infty(\N{r-1}\backslash \GL_{r-1}(F),\psi_{r-1}^{-1})$ the space of locally constant functions
$f:\GL_{r-1}(F)\to\mathbb C$ such that $f(ug)=\psi_{r-1}^{-1}(u)f(g)$ for all $u\in \N{r-1}$ and whose support is compact
modulo left $\N{r-1}$.
Equivalently, this is the compact induction $\cInd_{\N{r-1}}^{\GL_{r-1}(F)}(\psi_{r-1}^{-1})$.
\end{definition}

The following extension statement is the key representation-theoretic input.

\begin{theorem}[Gelfand--Kazhdan extension]\label{thm:GK-extension}
Let $F$ be a non-archimedean local field.
Let $\Sigma$ be a generic irreducible admissible representation of $\GL_r(F)$, realized in its Whittaker model
$\W(\Sigma,\psi_r^{-1})$.
Then the restriction map
\[
\mathrm{Res}_\iota:\W(\Sigma,\psi_r^{-1})\to C^\infty(\GL_{r-1}(F)),\qquad
W\longmapsto \bigl(g\mapsto W(\iota(g))\bigr),
\]
has image containing $C_c^\infty(\N{r-1}\backslash \GL_{r-1}(F),\psi_{r-1}^{-1})$.
In particular, for every $f\in \cInd_{\N{r-1}}^{\GL_{r-1}(F)}(\psi_{r-1}^{-1})$ there exists
$W\in \W(\Sigma,\psi_r^{-1})$ such that $W(\iota(g))=f(g)$ for all $g\in \GL_{r-1}(F)$.
\end{theorem}

\begin{proof}[Proof (reference)]
This is a classical statement in the theory of Kirillov models.
For $p$-adic $F$ it is proved by Gelfand--Kazhdan \cite{GK75}; see also the discussion and archimedean analogs in
Jacquet and Kemarsky.
\end{proof}

\section{A compactly supported universal test function on $\GL_n$}

Fix $n\ge 1$.
For $m\ge 0$, define a function $f_m:\GL_n(F)\to \mathbb C$ by
\begin{equation}\label{eq:def-fm}
f_m(uk)=\psi_n^{-1}(u)\quad (u\in \N{n},\ k\in K_1(\pp^m)),\qquad
f_m(g)=0\quad \text{if } g\notin \N{n}K_1(\pp^m).
\end{equation}

\begin{lemma}\label{lem:fm-well}
The function $f_m$ is well-defined and belongs to $\cInd_{\N{n}}^{\GL_n(F)}(\psi_n^{-1})$.
Moreover, $\N{n}\cap K_1(\pp^m)=\N{n}(\oo)$.
\end{lemma}

\begin{proof}
If $uk=u'k'$ with $u,u'\in \N{n}$ and $k,k'\in K_1(\pp^m)$, then $k^{-1}k'=u^{-1}u'\in \N{n}\cap K_1(\pp^m)$.
Since $K_1(\pp^m)\subseteq \GL_n(\oo)$, any element of $\N{n}\cap K_1(\pp^m)$ is an upper-triangular unipotent matrix with
entries in $\oo$, i.e.\ lies in $\N{n}(\oo)$.
Conversely $\N{n}(\oo)\subseteq K_1(\pp^m)$ because elements of $\N{n}$ have last row $e_n$.
Thus $\N{n}\cap K_1(\pp^m)=\N{n}(\oo)$.

Because $\psi$ has conductor $\oo$, it is trivial on $\oo$, hence $\psi_n$ is trivial on $\N{n}(\oo)$.
Therefore $\psi_n^{-1}(u)=\psi_n^{-1}(u')$ and the assignment \eqref{eq:def-fm} is well-defined.

The support of $f_m$ modulo $\N{n}$ is contained in $K_1(\pp^m)$, which is compact.
Also $f_m$ is right $K_1(\pp^m)$-invariant, hence locally constant.
Thus $f_m\in \cInd_{\N{n}}^{\GL_n(F)}(\psi_n^{-1})$.
\end{proof}

\section{Lifting $f_m$ to a Whittaker function on $\GL_{n+1}$}

Let $\Pi$ be a generic irreducible admissible representation of $\GL_{n+1}(F)$, realized in $\W(\Pi,\psi_{n+1}^{-1})$.

\begin{proposition}\label{prop:Wm-exists}
For each $m\ge 0$ there exists $W_m\in \W(\Pi,\psi_{n+1}^{-1})$ such that
\[
W_m(\iota(g))=f_m(g)\qquad \text{for all }g\in \GL_n(F),
\]
where $f_m$ is as in \eqref{eq:def-fm}.
\end{proposition}

\begin{proof}
By Lemma~\ref{lem:fm-well}, $f_m\in \cInd_{\N{n}}^{\GL_n(F)}(\psi_n^{-1})$.
Apply Theorem~\ref{thm:GK-extension} with $r=n+1$ and $\Sigma=\Pi$ to obtain $W_m\in \W(\Pi,\psi_{n+1}^{-1})$
with $W_m(\iota(g))=f_m(g)$ for all $g$.
\end{proof}

\section{Newforms on $\GL_n$}

Let $\pi$ be a generic irreducible admissible representation of $\GL_n(F)$.
Its conductor ideal is $\qq(\pi)=\pp^{a(\pi)}$ for an integer $a(\pi)\ge 0$.

\begin{proposition}[Whittaker newforms]\label{prop:newform}
Let $\pi$ be generic irreducible admissible of $\GL_n(F)$ with conductor ideal $\qq(\pi)=\pp^{a(\pi)}$.
Then there exists a Whittaker function $V\in \W(\pi,\psi_n)$ which is right $K_1(\pp^{a(\pi)})$-invariant and unique up to scalars.
Normalizing by $V(1)=1$, one has $V(k)=1$ for all $k\in K_1(\pp^{a(\pi)})$.
\end{proposition}

\begin{proof}
This is standard local newform theory for $\GL_n$ (existence of a unique newvector fixed by $K_1(\qq(\pi))$)
together with the uniqueness of Whittaker models; see \cite{JPSS81,JPSS83} and the explicit Whittaker newform results of Miyauchi \cite{Miyauchi14}.
If $V$ is right $K_1(\pp^{a(\pi)})$-invariant, then for $k\in K_1(\pp^{a(\pi)})$ we have $V(k)=V(1\cdot k)=V(1)=1$.
\end{proof}

\section{Main theorem: universal conductor-bounded nonvanishing}

\begin{lemma}\label{lem:psi-constant}
Let $m\ge 0$ and $a\ge 0$.
Let $Q\in F^\times$ satisfy $Q\in \pp^{-a}$.
If $m\ge a$ and $k\in K_1(\pp^m)$, then
\[
\psi(Q\,k_{n,n})=\psi(Q).
\]
\end{lemma}

\begin{proof}
For $k\in K_1(\pp^m)$, we have $k_{n,n}\equiv 1\pmod{\pp^m}$, so $k_{n,n}=1+t$ with $t\in \pp^m$.
Then $Qk_{n,n}=Q+Qt$ with $Qt\in \pp^{m-a}\subseteq \oo$ because $m\ge a$.
Since $\psi$ is trivial on $\oo$, we get $\psi(Q+Qt)=\psi(Q)$.
\end{proof}

\begin{theorem}\label{thm:main}
Let $n\ge 1$, and let $\Pi$ be a generic irreducible admissible representation of $\GL_{n+1}(F)$ realized in $\W(\Pi,\psi_{n+1}^{-1})$.
For each $m\ge 0$, let $W_m\in \W(\Pi,\psi_{n+1}^{-1})$ be as in Proposition~\ref{prop:Wm-exists}.

Let $\pi$ be any generic irreducible admissible representation of $\GL_n(F)$ with conductor ideal $\qq(\pi)=\pp^{a(\pi)}$.
Assume $a(\pi)\le m$ and let $Q\in F^\times$ be a generator of $\qq(\pi)^{-1}=\pp^{-a(\pi)}$.
Let $V\in \W(\pi,\psi_n)$ be the normalized Whittaker newform of Proposition~\ref{prop:newform}.

Then the shifted Rankin--Selberg integral \eqref{eq:RS} is absolutely convergent for every $s\in\mathbb C$ and equals the nonzero constant
\[
Z(W_m,V,s;Q)=\psi^{-1}(Q)\cdot \mathrm{vol}\!\bigl(\N{n}(\oo)\backslash K_1(\pp^m)\bigr),
\]
hence is finite and nonzero for all $s\in\mathbb C$.
\end{theorem}

\begin{proof}
By Lemma~\ref{lem:conj} and Proposition~\ref{prop:Wm-exists},
\[
W_m(\iota(g)u_Q)=\psi^{-1}(Q g_{n,n})\,W_m(\iota(g))
=\psi^{-1}(Q g_{n,n})\,f_m(g).
\]
Thus the integrand in \eqref{eq:RS} is supported on $\N{n}K_1(\pp^m)$.
On this support we have $g\in \GL_n(\oo)$, hence $\abs{\det g}=1$, so $\abs{\det g}^{s-\frac12}\equiv 1$ and the integrand is independent of $s$.
Absolute convergence for every $s$ follows because $\N{n}\backslash \N{n}K_1(\pp^m)$ is compact.

Write $g=uk$ with $u\in \N{n}$ and $k\in K_1(\pp^m)$.
Since the last row of $u$ is $e_n$, we have $(uk)_{n,n}=k_{n,n}$, hence
\[
W_m(\iota(g)u_Q)=\psi^{-1}(Q k_{n,n})\,\psi_n^{-1}(u).
\]
Because $a(\pi)\le m$ and $Q\in \pp^{-a(\pi)}$, Lemma~\ref{lem:psi-constant} gives $\psi(Qk_{n,n})=\psi(Q)$, so
\[
W_m(\iota(g)u_Q)=\psi^{-1}(Q)\,\psi_n^{-1}(u)\qquad (g=uk\in \N{n}K_1(\pp^m)).
\]

Now $f_m$ is left $(\N{n},\psi_n^{-1})$-equivariant and $V$ is left $(\N{n},\psi_n)$-equivariant, so the product
$f_m(g)V(g)$ is left $\N{n}$-invariant.
Using Lemma~\ref{lem:fm-well} and the standard identification
\[
\N{n}\backslash \N{n}K_1(\pp^m)\ \simeq\ (\N{n}\cap K_1(\pp^m))\backslash K_1(\pp^m)
\ =\ \N{n}(\oo)\backslash K_1(\pp^m),
\]
we obtain
\[
Z(W_m,V,s;Q)=\psi^{-1}(Q)\int_{\N{n}(\oo)\backslash K_1(\pp^m)} V(k)\,dk.
\]
Finally, since $a(\pi)\le m$ we have $K_1(\pp^m)\subseteq K_1(\pp^{a(\pi)})$, hence by Proposition~\ref{prop:newform} the normalized newform satisfies
$V(k)=1$ for all $k\in K_1(\pp^m)$.
Therefore
\[
Z(W_m,V,s;Q)=\psi^{-1}(Q)\cdot \mathrm{vol}\!\bigl(\N{n}(\oo)\backslash K_1(\pp^m)\bigr),
\]
which is a nonzero constant (the volume is a strictly positive real number).
\end{proof}

\begin{corollary}\label{cor:resolution}
For every generic irreducible admissible $\pi$ of $\GL_n(F)$ (with any conductor exponent $a(\pi)$),
taking $m=a(\pi)$ and $W=W_{a(\pi)}$ as in Theorem~\ref{thm:main}, and taking $V$ to be the normalized Whittaker newform,
the integral \eqref{eq:RS} is finite and nonzero for all $s\in\mathbb C$ and for every generator $Q$ of $\qq(\pi)^{-1}$.
\end{corollary}

\begin{remark}[Dependence on Haar measures]
The numerical value of $\mathrm{vol}(\N{n}(\oo)\backslash K_1(\pp^m))$ depends on Haar normalizations, but it is always $>0$.
Hence the nonvanishing asserted in Theorem~\ref{thm:main} is independent of the choice of measures.
\end{remark}

\begin{remark}[Uniformity in $\pi$ up to conductor bound]
For fixed $\Pi$ and fixed $m$, the Whittaker function $W_m$ works simultaneously for \emph{all} generic $\pi$ of $\GL_n(F)$ with
conductor exponent $a(\pi)\le m$ (and all generators $Q$ of $\qq(\pi)^{-1}$).
\end{remark}
